package main

import (
	"math/rand"
	"time"
)

func main() {
	pen := NewPen(500, 500)
	rand.Seed(time.Now().UnixNano())

	pen.SetHeading(0)
	pen.SetPosition(100, 100)
	dist := 300.00
	for dist > 10 {
		r := rand.Intn(256)
		g := rand.Intn(256)
		b := rand.Intn(256)
		pen.Walk(dist)
		dist -= 3
		pen.Right(90)
		pen.SetRGB(float64(r), float64(g), float64(b))
	}
	pen.SavePNG("tree.png")
}
