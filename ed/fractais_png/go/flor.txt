package main

func fractal(pen *Pen, tamanho float64, nivel int) {
	if nivel == 0 {
		return
	}
	pen.Walk(tamanho)
	pen.Left(144)
	for i := 0; i < 5; i++ {
		fractal(pen, tamanho*0.34, nivel-1)
		pen.Right(72)
	}
	pen.Left(216)

	pen.Walk(-tamanho)
}

func main() {
	pen := NewPen(1200, 1200)
	pen.SetHeading(55)
	pen.SetPosition(600, 600)

	for i := 0; i < 5; i++ {
		fractal(pen, 300, 6)
		pen.Right(72)
	}

	pen.SavePNG("tree.png")
}
