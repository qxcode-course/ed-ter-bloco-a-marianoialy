package main

func desenhaTriangulo(pen *Pen, lado float64) {
	for i := 0; i < 3; i++ {
		pen.Walk(lado)
		pen.Right(120)
	}
}

func fractal(pen *Pen, r float64, nivel int) {
	if nivel == 0 {
		return
	}
	desenhaTriangulo(pen, r)

	for i := 0; i < 3; i++ {
		fractal(pen, r*0.5, nivel-1)
		pen.Up()
		pen.Walk(r)
		pen.Right(120)

		pen.Down()

	}
}

func main() {
	pen := NewPen(1500, 1500)

	pen.SetHeading(0)
	pen.SetPosition(300, 300)

	fractal(pen, 1000, 10)

	pen.SavePNG("tree.png")
}
